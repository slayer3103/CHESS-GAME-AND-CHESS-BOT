[Welcome Screen] 
      ↓ “Start Game” 

[Choose Opponent Screen]
      ↓ action == “game” && opponent == "human"
      → set_game_status(True, "human")
      → run_game_screen("human", difficulty=None)
      ↓ action == “game” && opponent == "computer"

[Difficulty Screen]
      ↓ action in {"easy","medium","hard"} 
      → set_game_status(True, "computer")
      → run_game_screen("computer", difficulty=<chosen>)
      ↓ action == “back”
      ← back to [Choose Opponent Screen]
      ↓ action == “welcome”
      ← back to [Welcome Screen]
